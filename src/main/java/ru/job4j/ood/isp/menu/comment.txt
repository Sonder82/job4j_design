@Override
    public Optional<MenuItemInfo> select(String itemName) {
        Optional<ItemInfo> itemInfo = findItem(itemName);
        MenuItemInfo menuItemInfo = new MenuItemInfo(itemInfo.get().getMenuItem(), itemInfo.get().getNumber());
